42def convert_doxygen_xml_to_lobster_common_interchange_format(doxygen_xml_folder: str, output_file_name: str) -> Ret: 

43 # lobster-trace: SwRequirements.sw_req_cli_doxygen_xml_folder 

44 # lobster-trace: SwRequirements.sw_req_output_file_format 

45 # lobster-trace: SwRequirements.sw_req_cli_output 

46 # lobster-trace: SwRequirements.sw_req_no_trace 

47 """Convert xml files in doxygen_xml_folder to LOBSTER common interchange format file with name 

48 output. 

49 

50 Args: 

51 doxygen_xml_folder (str): The Doxygen XML output directory, where the file index.xml is located. 

52 output_file_name (str): Path and file name for LOBSTER common interchange format file. 

53 

54 Return: 

55 Ret.RET_OK: LOBSTER common interchange format file successful created. 

56 Ret.RET_ERROR_FILEPATH_INVALID: No index.xml file in doxygen_folder 

57 Ret.RET_ERROR: Conversion not successful. 

58 """ 

59 ret_status = Ret.RET_ERROR 

60 is_index_file_found = False 

61 

62 if not os.path.isfile(doxygen_xml_folder + "/index.xml"): 

63 LOG.print_error( 

64 f"No doxygen index.xml file in doxygen_xml_folder {doxygen_xml_folder}.") 

65 ret_status = Ret.RET_ERROR_FILEPATH_INVALID 

66 is_index_file_found = False 

67 else: 

68 is_index_file_found = True 

69 

70 if is_index_file_found: 

71 lobster_items = get_lobster_items_from_doxygen_xml_folder( 

72 doxygen_xml_folder) 

73 # Continue only if no error during parsing. 

74 if lobster_items is not None: 

75 # Check if lobster items are found. 

76 if 0 == len(lobster_items): 

77 LOG.print_warning( 

78 "No lobster items found in the doxygen XML output.") 

79 

80 if rule_check(lobster_items) is True: 

81 write_lobster_common_interchange_format_file( 

82 lobster_items, output_file_name) 

83 ret_status = Ret.RET_OK 

84 

85 return ret_status
